ajax封装

发表于 2017-07-03 | 分类于 前端技术 | 阅读次数

平常jquery用的不少，但是有时候只是需要用ajax，从而引入jquery就显得有些臃肿，这里简单封装一下ajax函数，方便以后使用～～～

/* 封装ajax函数
       * @param {string}methods http连接的方式，包括POST和GET两种方式
       * @param {string}url     发送请求的url
       * @param {object}data    发送的参数，对象类型
       * @param {function}cb    ajax发送并接收成功调用的回调函数
       * @param {boolean}async  是否为异步请求，true为异步，false为同步
*/

function ajax(methods,url,data,cb,async){

    var xhr = null

    try{
        xhr = new XMLHttpRequest()
    }catch(e){
        xhr = new ActiveXObject('Microsodt.XMLHTTP')
    }                

    if(methods == 'get' && data){
        url += '?' + data
    }

    xhr.open(methods,url,async)    

    if(methods == 'get'){
        xhr.send()
    }else{
        var str = ''
         for (var key in data){
             str += key + "=" + data[key] + "&"
         }
        xhr.setRequsetHeader('Content-Type", "application/x-www-form-urlencoded')
        xhr.send(str)
    }

    xhr.onreadystatechange = function(){

        if(xhr.readystate == 4 && xhr.status == 400){
            cb : cb(xhr.responseText)
        }else{
            console.log('ERROR:' + xhr.status)
        }

    }

}

调用方法如下：

ajax('get','getData.php','',function(data){
    var data = JSON.parse(data)
    console.log(data)
},true)
